document.addEventListener('DOMContentLoaded', function() {
    const menuToggle = document.querySelector('.menu-toggle');
    const mainNav = document.querySelector('.main-nav');
    const body = document.body;

    // メニュートグルの動作
    menuToggle.addEventListener('click', function() {
        menuToggle.classList.toggle('active');
        mainNav.classList.toggle('active');
        body.classList.toggle('menu-open');
    });

    // メニューリンクをクリックしたときの動作
    const menuLinks = document.querySelectorAll('.main-nav a');
    menuLinks.forEach(link => {
        link.addEventListener('click', function() {
            // 現在のページのリンクをクリックした場合は、メニューを閉じるだけ
            if (this.classList.contains('active')) {
                menuToggle.classList.remove('active');
                mainNav.classList.remove('active');
                body.classList.remove('menu-open');
                return;
            }
            
            // 他のページへのリンクの場合は、通常の遷移を行う
            menuToggle.classList.remove('active');
            mainNav.classList.remove('active');
            body.classList.remove('menu-open');
        });
    });

    // スクロール時のメニューバーの挙動
    let lastScrollTop = 0;
    window.addEventListener('scroll', function() {
        const currentScrollTop = window.pageYOffset || document.documentElement.scrollTop;
        const header = document.querySelector('.header-content');
        
        // スクロール方向に応じてヘッダーの表示/非表示を切り替え
        if (currentScrollTop > lastScrollTop) {
            // 下スクロール時
            header.style.transform = 'translateY(-100%)';
        } else {
            // 上スクロール時
            header.style.transform = 'translateY(0)';
        }
        lastScrollTop = currentScrollTop;
    });

    // 画面サイズ変更時の処理
    window.addEventListener('resize', function() {
        if (window.innerWidth > 768) {
            // PCサイズになった時にメニューを初期状態に戻す
            mainNav.classList.remove('active');
            menuToggle.classList.remove('active');
            body.classList.remove('menu-open');
            document.querySelector('.header-content').style.transform = 'translateY(0)';
        }
    });

    // タッチスワイプでのメニュー操作（スマートフォン用）
    let touchStartX = 0;
    let touchEndX = 0;
    
    document.addEventListener('touchstart', function(e) {
        touchStartX = e.changedTouches[0].screenX;
    }, false);

    document.addEventListener('touchend', function(e) {
        touchEndX = e.changedTouches[0].screenX;
        handleSwipe();
    }, false);

    function handleSwipe() {
        const swipeThreshold = 100; // スワイプを検知する閾値
        
        if (touchEndX < touchStartX - swipeThreshold) {
            // 左スワイプでメニューを閉じる
            if (mainNav.classList.contains('active')) {
                menuToggle.classList.remove('active');
                mainNav.classList.remove('active');
                body.classList.remove('menu-open');
            }
        } else if (touchEndX > touchStartX + swipeThreshold) {
            // 右スワイプでメニューを開く
            if (!mainNav.classList.contains('active')) {
                menuToggle.classList.add('active');
                mainNav.classList.add('active');
                body.classList.add('menu-open');
            }
        }
    }

    function debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    }

    // スクロール処理の最適化

        const deltaX = touchEndX - touchStartX;
        const deltaY = Math.abs(touchEndY - touchStartY);

        // 水平スワイプの判定
        if (Math.abs(deltaX) > deltaY && Math.abs(deltaX) > 50) {
            e.preventDefault();
            if (deltaX < 0) {
                closeMenu();
            }
        }
    }, { passive: false });

        menuToggle.classList.remove('active');
        mainNav.classList.remove('active');
        body.classList.remove('menu-open');
        menuToggle.setAttribute('aria-expanded', 'false');
        mainNav.setAttribute('aria-hidden', 'true');
    }
